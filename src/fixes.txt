// Enhanced validation for real avatars (stricter for 2025 patterns)
function isRealAvatar(url) {
  if (!url) {
    console.log('[Avatar Debug] isRealAvatar: URL is empty or null');
    return false;
  }
  
  // Invalid patterns (placeholders/defaults)
  const invalidPatterns = [
    'transparent', 'placeholder', 'default', 'blank', 'empty',
    'group-avatar-icons', 'fbcdn.net/rsrc.php', // Static assets
    'static.xx.fbcdn.net', // Static FB resources
    'group_default', 'default_group_icon', // Common default avatar patterns
    '.svg', '.gif?cid=', '/p40x40/', '/c0.0.', // Small placeholders
    'data:image/gif;base64,R0lGODlhAQAB', // Transparent GIF
    'data:image/svg+xml', // Default SVGs
    '/images/defaults/', '/placeholders/' // Additional placeholder paths
  ];
  
  if (invalidPatterns.some(pattern => url.toLowerCase().includes(pattern))) {
    console.log('[Avatar Debug] Rejected as invalid/placeholder:', url);
    return false;
  }
  
  // Valid patterns (real avatars) - stricter to ensure actual group images
  const validPatterns = [
    /^https:\/\/scontent[^\/]*\.fna\.fbcdn\.net\/v\/t\d+\.\d+-\d+\/.*\.(jpg|jpeg|png|webp)/i, // Versioned FB CDN
    /^https:\/\/scontent.*\.xx\.fbcdn\.net\/v\/t\d+\.\d+-\d+\/.*\.(jpg|jpeg|png|webp)/i, // Alternative CDN
    /^https:\/\/platform-lookaside\.fbsbx\.com\/.*\.(jpg|jpeg|png|webp)/i, // Platform CDN
    /^data:image\/(jpeg|png|webp);base64,[A-Za-z0-9+/=]{500,}/ // Real base64 (increased size threshold)
  ];
  
  if (validPatterns.some(regex => regex.test(url))) {
    console.log('[Avatar Debug] Accepted as real avatar:', url);
    return true;
  }
  
  console.log('[Avatar Debug] Uncertain URL - rejected:', url);
  return false;
}

// Enhanced avatar extraction function for Facebook groups (2025 compatible)
function findImage(node) {
  if (!node) {
    console.log('[Avatar Debug] findImage called with null/undefined node');
    return '';
  }
  
  console.log('[Avatar Debug] findImage started on node:', {
    tag: node.tagName,
    classes: node.className?.substring(0, 50) || 'none',
    id: node.id || 'none',
    visible: node.offsetParent !== null
  });

  const strategies = [
    // Strategy 1: Direct <img> with common FB avatar classes
    {
      name: 'Direct img with FB classes',
      fn: (targetNode = node) => {
        const selectors = [
          'img[class*="x1rg5ohu"][alt*="group"]', // Avatar-specific class + group alt
          'img[class*="xt0psk2"][alt*="group"]', // Profile pic class + group alt
          'img[class*="x5yr21d"][data-visualcompletion="media-vc-image"]', // Thumbnail with visual completion
          'img[class*="xh8yej3"][alt*="perfil"]', // Image container with profile alt
          'img[alt*="group"], img[alt*="perfil"]' // Any image with group/profile alt
        ];
        for (const sel of selectors) {
          const img = targetNode.querySelector(sel);
          if (img) {
            let src = img.src || img.getAttribute('data-src') || ''; // Handle lazy load
            if (src && isRealAvatar(src)) {
              console.log(`[Avatar Debug] Found via ${this.name} selector ${sel}:`, src);
              return src;
            }
          }
        }
        return null;
      }
    },
    
    // Strategy 2: Any <img> with CDN src, prioritizing avatars
    {
      name: 'Any img with CDN src',
      fn: (targetNode = node) => {
        const img = targetNode.querySelector('img[src^="https://scontent"], img[data-src^="https://scontent"]');
        if (img) {
          let src = img.src || img.getAttribute('data-src') || '';
          if (src && isRealAvatar(src)) {
            console.log(`[Avatar Debug] Found via ${this.name}:`, src);
            return src;
          }
        }
        return null;
      }
    },
    
    // Strategy 3: Background-image on divs, specific to avatars
    {
      name: 'Background image',
      fn: (targetNode = node) => {
        const bg = targetNode.querySelector('[style*="background-image"][class*="x1rg5ohu"], [style*="background-image"][class*="group"]');
        if (bg && bg.style.backgroundImage) {
          const urlMatch = bg.style.backgroundImage.match(/url\("?(.*?)"?\)/);
          const url = urlMatch ? urlMatch[1] : '';
          if (url && isRealAvatar(url)) {
            console.log(`[Avatar Debug] Found via ${this.name}:`, url);
            return url;
          }
        }
        return null;
      }
    },
    
    // Strategy 4: SVG <image> (strict validation)
    {
      name: 'SVG image',
      fn: (targetNode = node) => {
        const svgImages = targetNode.querySelectorAll('svg image');
        for (let svgImage of svgImages) {
          let src = svgImage.getAttributeNS('http://www.w3.org/1999/xlink', 'href') ||
                    svgImage.getAttribute('xlink:href') ||
                    svgImage.getAttribute('href') ||
                    '';
          if (src && isRealAvatar(src)) {
            console.log(`[Avatar Debug] Found via ${this.name}:`, src);
            return src;
          } else if (src) {
            console.log(`[Avatar Debug] SVG found but rejected:`, src.substring(0, 50));
          }
        }
        return null;
      }
    },
    
    // Strategy 5: Canvas data URL (rare, with strict validation)
    {
      name: 'Canvas fallback',
      fn: (targetNode = node) => {
        const canvas = targetNode.querySelector('canvas');
        if (canvas) {
          try {
            const dataUrl = canvas.toDataURL('image/jpeg', 0.8); // Use JPEG to reduce size
            if (dataUrl && isRealAvatar(dataUrl)) {
              console.log(`[Avatar Debug] Found via ${this.name} (canvas export):`, dataUrl.substring(0, 50));
              return dataUrl;
            }
          } catch (e) {
            console.log(`[Avatar Debug] Canvas access error:`, e.message);
          }
        }
        return null;
      }
    }
  ];

  // Execute strategies on current node
  for (const strat of strategies) {
    const result = strat.fn();
    if (result) return result;
    console.log(`[Avatar Debug] Strategy "${strat.name}" returned null`);
  }

  // Parent traversal (increased to 6 levels for deeper nesting)
  let parent = node.parentElement;
  let level = 0;
  while (parent && level < 6) {
    level++;
    console.log(`[Avatar Debug] Traversing parent level ${level}`);
    for (const strat of strategies) {
      const result = strat.fn(parent);
      if (result) return result;
    }
    parent = parent.parentElement;
  }

  // Sibling check as final fallback
  console.log('[Avatar Debug] Checking siblings');
  const siblings = node.parentElement?.children || [];
  for (const sibling of siblings) {
    if (sibling !== node) {
      for (const strat of strategies) {
        const result = strat.fn(sibling);
        if (result) return result;
      }
    }
  }

  console.log('[Avatar Debug] No avatar found after all attempts');
  console.log('[Avatar Debug] Node outerHTML sample:', node.outerHTML.substring(0, 300) + '...');
  return '';
}

// Updated avatar extraction with multiple container strategies
function extrairAvatarGrupo(elemento) {
  console.log('[Avatar Debug] Starting extrairAvatarGrupo');
  
  // Wait for potential lazy-loaded images
  async function waitForImages(container) {
    if (!container) return false;
    const img = container.querySelector('img[data-src], img[src]');
    if (img && (img.src || img.getAttribute('data-src'))) {
      return true;
    }
    return new Promise(resolve => {
      const observer = new MutationObserver(() => {
        const updatedImg = container.querySelector('img[data-src], img[src]');
        if (updatedImg && (updatedImg.src || updatedImg.getAttribute('data-src'))) {
          observer.disconnect();
          resolve(true);
        }
      });
      observer.observe(container, { attributes: true, childList: true, subtree: true });
      setTimeout(() => {
        observer.disconnect();
        resolve(false);
      }, 3000); // 3-second timeout for image loading
    });
  }

  const containerStrategies = [
    () => elemento.closest('div[role="article"]'),
    () => elemento.closest('div[data-pagelet*="GroupsTab"]'),
    () => elemento.closest('div[class*="x1yztbdb"]') || elemento.closest('div[class*="group"]'),
    () => elemento.closest('div[class*="x1n2onr6"]') || elemento.closest('div[class*="x1ja2u2z"]'),
    () => {
      let parent = elemento.parentElement;
      let depth = 0;
      while (parent && depth < 6) {
        if (parent.querySelector('img[alt*="group"], img[alt*="perfil"]')) return parent;
        parent = parent.parentElement;
        depth++;
      }
      return null;
    },
    () => elemento.parentElement,
    () => elemento.parentElement?.parentElement
  ];

  let avatar = '';
  let strategyIndex = 0;

  for (const getContainer of containerStrategies) {
    strategyIndex++;
    const container = getContainer();
    if (container) {
      console.log(`[Avatar Debug] Trying container strategy ${strategyIndex}`);
      // Wait for images to load in this container
      const imagesLoaded = await waitForImages(container);
      if (!imagesLoaded) {
        console.log(`[Avatar Debug] No images loaded for strategy ${strategyIndex}`);
        continue;
      }
      avatar = findImage(container);
      if (avatar) {
        console.log('[Avatar Debug] Success with strategy', strategyIndex);
        return avatar;
      }
    } else {
      console.log(`[Avatar Debug] Strategy ${strategyIndex} returned null container`);
    }
  }

  console.log('[Avatar Debug] No avatar found');
  return '';
}
